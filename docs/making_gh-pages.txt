Requires asciidoctor

Source in docs/

To update:

  git add $(find . -name \*.adoc)
  rm -f $(find . -name \*.html)
  git commit
  git push
  git branch -f gh-pages <branch>
     <branch> is whatever is to be used, usually master
  git checkout gh-pages
  make
  git add -f $(find . -name \*.html)
  git commit -m Update
  git push --force
